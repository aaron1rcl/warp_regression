df = fetch_bitcoin_daily()
dates = pd.to_datetime(df['date'])
y = log_price(df['price_usd'].to_numpy())
n = len(y)
t_idx = day_index(n)
t = normalized_time(n)
split = split_bitcoin_holdout(n, test_days=365, dates=dates)
train_idx, test_idx = split['train_idx'], split['test_idx']
n_train = split['n_train']
y_tr, t_idx_tr, t_tr = y[train_idx], t_idx[train_idx], t[train_idx]
dates_tr = dates[train_idx]
print(f'train n={n_train} test n={split["n_test"]}')

train n=5406 test n=365

B, C, z, trend, _ = fit_log_trend(y_tr, t_idx_tr)
residual = y_tr - trend
sine_fit = fit_sine_peak_presize(residual, y_tr, t_tr, dates_tr)
z_full = _sine_from_fit(t, sine_fit)
sine_fit_full = {**sine_fit, 'z': z_full[:n_train]}
print(f'presize corr={sine_fit["presize_corr"]:.3f}')

EPOCHS = 24000
model = WarpModel(WarpModelConfig(readout='log_trend_sine', n_knots=8, epochs=EPOCHS, fit_lambda=0.9, lr=0.2, seed=0))
fit_res = model.fit(y_tr, t_idx=t_idx_tr, t_norm=t_tr, sine_fit=sine_fit_full)
print(f'train R²={fit_res.r2:.3f} rmse={fit_res.rmse:.3f}')
